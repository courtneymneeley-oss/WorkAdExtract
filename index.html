Setup guide
One-time setup — takes about 3 minutes:
1. Open your Google Sheet
2. Click Extensions → Apps Script
3. Delete any existing code and paste the script below
4. Click Deploy → New deployment
5. Type: Web app — set "Who has access" to Anyone
6. Click Deploy, copy the URL, paste it above
APPS SCRIPT CODE — copy all of this:
function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getActiveSheet(); if (sheet.getLastRow() === 0) { var headers = [ '会社名','住所','TEL','職種・タイトル', '雇用形態','求人内容','掲載日' ]; sheet.appendRow(headers); var hRange = sheet.getRange(1, 1, 1, headers.length); hRange.setFontWeight('bold'); hRange.setBackground('#f3f3f3'); } var telCell = data.phone || ''; if (data.contact) telCell += '\n担当: ' + data.contact; if (data.contact_phone) telCell += '\n' + data.contact_phone; var contentCell = ''; if (data.categories && data.categories.length > 0) { contentCell = data.categories.map(function(c) { return c.name + '\n' + c.content; }).join('\n\n'); } var row = [ data.company || '', data.address || '', telCell, data.title || '', data.contract || '', contentCell, data.date || '' ]; sheet.appendRow(row); var newRow = sheet.getLastRow(); var lastCol = row.length; sheet.getRange(newRow, 1, 1, lastCol).setVerticalAlignment('top'); sheet.getRange(newRow, 1, 1, lastCol).setWrap(true); if (data.categories && data.categories.length > 0) { var contentCol = 6; var rt = SpreadsheetApp.newRichTextValue(); var fullText = contentCell; rt.setText(fullText); var pos = 0; data.categories.forEach(function(c) { var nameLen = c.name.length; rt.setTextStyle(pos, pos + nameLen, SpreadsheetApp.newTextStyle().setBold(true).build() ); pos += nameLen + 1 + (c.content ? c.content.length : 0) + 2; }); sheet.getRange(newRow, contentCol).setRichTextValue(rt.build()); } return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:'error', message: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
